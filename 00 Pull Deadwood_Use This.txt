/*NO RESPONSE DEADWOOD*/
CREATE
or replace TABLE OMF_EXT_USER_SPACE.AJONES.NO_RESPONSE_DEADWOOD as
Select
    EMAIL_ADDR_TXT,
    DAYS_SINCE_OPEN,
    DAYS_SINCE_CLICK,
    DAYS_SINCE_SUBSCRIBE,
    to_varchar (CURRENT_DATE(), 'YYYY_MM_DD') as MOD_DATE
from
    PROD_OMF_INBOUND_DATA_SHARE.PROD_ANALYTIC_MART_SHARE.ZETA_DEADWOOD_SUPP
where
    DAYS_SINCE_OPEN is null
    AND DAYS_SINCE_CLICK is null
    AND DAYS_SINCE_SUBSCRIBE > 360
    /*CREATES FILE OF INVALID EMAILS AT ZETA - DOWNLOAD THIS FILE TO THE DECEASED LOCATION*/
    CREATE
    or replace TABLE OMF_EXT_USER_SPACE.AJONES.INVALID_ZETA_EMAILS as
select
    EMAIL_ADDR_TXT,
    to_varchar (CURRENT_DATE(), 'YYYY_MM_DD') as MOD_DATE
from
    PROD_OMF_INBOUND_DATA_SHARE.PROD_ANALYTIC_MART_SHARE.ZETA_DEADWOOD_SUPP
where
    VALIDSTATUS = '0';
