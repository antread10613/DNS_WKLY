CREATE
or replace TABLE OMF_EXT_USER_SPACE.AJONES.ZETA_UNSUBS AS
SELECT
    EMAIL_ADDR_TXT,
    to_varchar (CURRENT_DATE(), 'YYYY_MM_DD') as MOD_DATE
FROM
    PROD_OMF_INBOUND_DATA_SHARE.PROD_ANALYTIC_MART_SHARE.ZETA_DEADWOOD_SUPP;
WHERE
    UNSUBDATE IS NOT NULL;
CREATE
    or replace TABLE OMF_EXT_USER_SPACE.AJONES.EAUTH_CODES AS
SELECT
    A.EMAIL_ID,
    A.AUTH_CD,
    A.CUR_AUTH_FLG,
    B.EMAIL_ADDR_TXT
FROM
    PROD_OMF_INBOUND_DATA_SHARE.PROD_ANALYTIC_MART_SHARE.MDM_EMAIL_AUTHORIZATION AS A
    LEFT JOIN PROD_OMF_INBOUND_DATA_SHARE.PROD_ANALYTIC_MART_SHARE.ZETA_DEADWOOD_SUPP AS B ON A.EMAIL_ID = B.EMAIL_ID
WHERE
    a.CUR_AUTH_FLG = 'N'
    AND SUBSTR(AUTH_CD, 1, 2) = 'UN';
