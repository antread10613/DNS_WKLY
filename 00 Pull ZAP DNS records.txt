CREATE
or replace TABLE OMF_EXT_USER_SPACE.AJONES.ZAP_DNS AS
SELECT 	A.CUST_ID,
		B.AUTH_CD,  to_varchar (CURRENT_DATE(), 'YYYY_MM_DD') as MOD_DATE
FROM PROD_OMF_INBOUND_DATA_SHARE.PROD_ANALYTIC_MART_SHARE.ACCOUNT A 
INNER JOIN PROD_OMF_INBOUND_DATA_SHARE.PROD_ANALYTIC_MART_SHARE.MDM_IND_AUTHORIZATION B
ON A.INDIV_ID = B.INDIV_ID 
where  A.CUST_ID <> 0
OR A.CUST_ID is not null
and     B.CUR_AUTH_FLG = 'N' AND
                B.AUTH_CD IN ('DNS','DNS_MAIL','OM_DNM')
