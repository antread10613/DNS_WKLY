/*PC SPECTRUM RECORDS*/
CREATE
or replace TABLE OMF_EXT_USER_SPACE.AJONES.PC_SPECTRUM_FILE as
select
    DISTINCT INDIV_ID,
    CUST_ID,
    ACTIVE_SPECTRUM_YN,
    to_varchar (CURRENT_DATE(), 'YYYY_MM_DD') as MOD_DATE
from
    PROD_OMF_INBOUND_DATA_SHARE.PROD_ANALYTIC_MART_SHARE.CUSTOMER
where
    ACTIVE_SPECTRUM_YN in('Y')
    and cust_id is not null;

    
    /*FC SPECTRUM RECORDS*/
    CREATE
    or replace TABLE OMF_EXT_USER_SPACE.AJONES.FC_SPECTRUM AS
SELECT
    INDIV_ID,
    IND_DOJ_SUPP_FLG,
    TOT_FCL_YN,
    TOT_FCR_YN,
    to_varchar (CURRENT_DATE(), 'YYYY_MM_DD') as MOD_DATE
from
    PROD_OMF_INBOUND_DATA_SHARE.PROD_ANALYTIC_MART_SHARE.INDIVIDUAL
WHERE
    IND_DOJ_SUPP_FLG = 'Y'
    AND TOT_FCR_YN = 'Y'
    OR TOT_FCL_YN = 'Y';
CREATE
    or replace TABLE OMF_EXT_USER_SPACE.AJONES.FC_SPECTRUM_FILE AS
SELECT
    A.INDIV_ID,
    A.IND_DOJ_SUPP_FLG,
    B.CUST_ID,
    B.INDIV_ID AS INDIV_ID_CUST
FROM
    OMF_EXT_USER_SPACE.AJONES.FC_SPECTRUM A
    INNER JOIN PROD_OMF_INBOUND_DATA_SHARE.PROD_ANALYTIC_MART_SHARE.CUSTOMER B ON A.INDIV_ID = B.INDIV_ID
where
    b.cust_id is not null;
