CREATE
or replace TABLE OMF_EXT_USER_SPACE.AJONES.ZETA_RECORDS as
SELECT
    EMAIL_ADDR_TXT,
    DAYS_SINCE_OPEN,
    DAYS_SINCE_CLICK,
    to_varchar (CURRENT_DATE(), 'YYYY_MM_DD') as MOD_DATE
FROM
    PROD_OMF_INBOUND_DATA_SHARE.PROD_ANALYTIC_MART_SHARE.ZETA_DEADWOOD_SUPP
WHERE
    VALIDSTATUS <> 0
    OR UNSUBDATE IS NOT NULL;
UPDATE
    OMF_EXT_USER_SPACE.AJONES.ZETA_RECORDS
SET
    DAYS_SINCE_OPEN = -1
WHERE
    DAYS_SINCE_OPEN IS NULL;
UPDATE
    OMF_EXT_USER_SPACE.AJONES.ZETA_RECORDS
SET
    DAYS_SINCE_CLICK = -1
WHERE
    DAYS_SINCE_CLICK IS NULL;
