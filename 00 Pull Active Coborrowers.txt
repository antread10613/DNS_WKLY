CREATE
OR REPLACE TABLE OMF_EXT_USER_SPACE.AJONES.ACTIVE_COBOR AS
SELECT
    A.INDIV_ID,
    A.TOT_ACTIVE_COBOR_YN,
    B.CUST_ID,
    B.CUST_ID_OM
FROM
    PROD_OMF_INBOUND_DATA_SHARE.PROD_ANALYTIC_MART_SHARE.INDIVIDUAL A
    INNER JOIN PROD_OMF_INBOUND_DATA_SHARE.PROD_ANALYTIC_MART_SHARE.CUSTOMER B ON A.INDIV_ID = B.INDIV_ID
WHERE
    TOT_ACTIVE_COBOR_YN = 'Y';
CREATE
    OR REPLACE TABLE OMF_EXT_USER_SPACE.AJONES.ACTIVE_COBOR_FILE AS
SELECT
    CUST_ID,
    TOT_ACTIVE_COBOR_YN,
    to_varchar (CURRENT_DATE(), 'YYYY_MM_DD') as MOD_DATE
FROM
    OMF_EXT_USER_SPACE.AJONES.ACTIVE_COBOR
where
    CUST_ID = CUST_ID_OM
    and CUST_ID <> null
    OR CUST_ID <> 0;
